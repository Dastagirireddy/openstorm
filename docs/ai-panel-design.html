3. Message System (LangChain-Inspired)

All communication uses typed message objects. No raw strings in the pipeline.

Message Types

// ═══════════════════════════════════════════════════════════════
// MESSAGE TYPES — Typed messages for all communication
// ═══════════════════════════════════════════════════════════════

/// Base message trait — all messages implement this
trait Message: Send + Sync {
    fn role(&self) -> &str;
    fn content(&self) -> &[ContentBlock];
    fn metadata(&self) -> &MessageMetadata;
    fn id(&self) -> &str;
}

/// System message — instructions for the model
struct SystemMessage {
    id: String,
    content: String,              // The system prompt
    metadata: MessageMetadata,
}

/// Human message — user input
struct HumanMessage {
    id: String,
    content: Vec<ContentBlock>,   // Text, images, files
    name: Option<String>,         // Optional user identifier
    metadata: MessageMetadata,
}

/// AI message — model output (may include tool calls)
struct AIMessage {
    id: String,
    content: Vec<ContentBlock>,   // Text + reasoning blocks
    tool_calls: Vec<ToolCall>,    // Tools the model wants to call
    usage: Option<UsageMetadata>,
    response_metadata: ResponseMetadata,
    metadata: MessageMetadata,
}

/// Tool message — result of a tool execution
struct ToolMessage {
    id: String,
    content: String,              // Tool output text
    tool_call_id: String,         // Matches AIMessage.tool_calls[].id
    tool_name: String,
    artifact: Option<Value>,      // Data not sent to LLM
    is_error: bool,
    metadata: MessageMetadata,
}

/// Agent message — inter-agent communication
struct AgentMessage {
    id: String,
    from_agent: String,
    to_agent: String,
    content: Vec<ContentBlock>,
    message_type: AgentMessageType,
    metadata: MessageMetadata,
}

/// Metadata attached to every message
struct MessageMetadata {
    timestamp: DateTime<Utc>,
    agent_id: Option<String>,
    session_id: String,
    token_count: Option<usize>,
    cost: Option<f64>,
}

Content Blocks

// ═══════════════════════════════════════════════════════════════
// CONTENT BLOCKS — Structured content within messages

enum ContentBlock {
    /// Plain text
    Text { text: String },

    /// Model reasoning
    Reasoning { reasoning: String, signature: Option<String> },

    /// Code block with language
    Code { language: String, code: String },

    /// File reference
    FileReference { path: String, line_range: Option<(u32, u32)> },

    /// Tool call (within AI message)
    ToolCallBlock { tool_call: ToolCall },

    /// Tool result (within Tool message)
    ToolResultBlock { result: String, is_error: bool },

    /// Diff (for code changes)
    Diff { file_path: String, hunks: Vec<DiffHunk> },

    /// Error
    Error { message: String, code: Option<String> },
}

/// Tool call the model wants to execute
struct ToolCall {
    id: String,               // Unique ID for this call
    name: String,             // Tool name
    args: Value,              // JSON arguments
}

/// Usage metadata
struct UsageMetadata {
    input_tokens: usize,
    output_tokens: usize,
    total_tokens: usize,
    cache_read_tokens: Option<usize>,
    estimated_cost: Option<f64>,
}

4. Tool System (LangChain-Inspired)

ToolRuntime — The Core Pattern

Inspired by LangChain
Tools don't access global state — they receive a ToolRuntime object with everything they need: state, context, store, stream writer. 
// ═══════════════════════════════════════════════════════════════
// TOOL RUNTIME — Injected into every tool call
// ═══════════════════════════════════════════════════════════════

struct ToolRuntime {
    /// Short-term memory: messages in current conversation
    state: Arc<Mutex<ConversationState>>,

    /// Immutable context: project path, model info, session ID
    context: ToolContext,

    /// Long-term memory: persistent across sessions
    store: Arc<dyn Store>,

    /// Stream writer: emit real-time updates during execution
    stream_tx: mpsc::Sender<StreamEvent>,

    /// Execution info: thread ID, run ID, attempt number
    execution: ExecutionInfo,

    /// Tool call ID: for correlating results with calls
    tool_call_id: String,
}

/// Immutable per-invocation context
struct ToolContext {
    project_path: PathBuf,
    model_name: String,
    session_id: String,
    agent_id: String,
    agent_role: AgentRole,
    max_result_chars: usize,
}

/// Conversation state (short-term memory)
struct ConversationState {
    messages: Vec<Box<dyn Message>>,
    current_plan: Option<Plan>,
    todo_items: Vec<TodoItem>,
    files_modified: Vec<PathBuf>,
    total_tool_calls: u32,
    total_tokens: usize,
}

trait Tool: Send + Sync {
    /// Tool name (snake_case, no spaces)
    fn name(&self) -> &str;

    /// Description shown to the LLM
    fn description(&self) -> &str;

    /// JSON Schema for input validation
    fn input_schema(&self) -> Value;

    /// Return directly to user without another LLM call?
    fn return_direct(&self) -> bool { false }
}

    execution_time_ms: u64,
    tool_call_id: String,

/// Tool categories for organization
enum ToolCategory {

Question Tool — LLM-to-User Input

Why a Tool?

The LLM shouldn't just "ask" in plain text. A structured question() tool forces the LLM to define options, lets the frontend render a UI, and returns a clean typed answer back to the LLM.

// ═══════════════════════════════════════════════════════════════

// ═══════════════════════════════════════════════════════════════

#[derive(Serialize, Deserialize)]
struct QuestionTool {
    questions: Vec<QuestionItem>,
}

#[derive(Serialize, Deserialize)]
struct QuestionItem {
    /// Unique ID for the answer key
    id: String,
    /// The question text (supports markdown)
    text: String,

    #[serde(default)]
    kind: QuestionKind,
    /// Options for choice/confirm types

    /// Whether user can select multiple options
    #[serde(default)]
    multiple: bool,
    /// Default value if user skips
    #[serde(default)]
    default: Option<String>,

#[derive(Serialize, Deserialize)]

Rules for the LLM

Prompt Rules (add to system prompt)
  • Use question() when: you need user input before proceeding, you're uncertain about approach, or you need confirmation for a destructive action
  • Don't use question() when: you can proceed without asking (just do it)
  • Batch questions: ask all independent questions in one call, don't ask sequentially
  • Provide options: always give concrete choices instead of open-ended "what do you want?"
  • One exception: if answer to Q2 depends on Q1, ask Q1 first, then Q2 in a new call

// ═══════════════════════════════════════════════════════════════
// AGENT LOOP — The main execution cycle
// ═══════════════════════════════════════════════════════════════

struct AgentRuntime {
    provider: Arc<dyn LLMProvider>,
    tools: Arc<ToolRegistry>,
    context_service: Arc<ContextService>,
    permission_service: Arc<PermissionService>,
    store: Arc<dyn Store>,
    event_tx: mpsc::Sender<AgentEvent>,
    config: AgentConfig,
}

struct AgentConfig {
    model: String,
    max_iterations: usize,          // Default: 20
    max_consecutive_text: usize,    // Default: 3
    context_token_budget: usize,    // Default: 8192
    tool_result_cache_ttl: Duration, // Default: 5 min
}

impl AgentRuntime {
    async fn run(
        &self,
        user_message: HumanMessage,
        history: Vec<Message>,
    ) -> Result<AgentEvent> {
        let mut messages = vec![];

        // 1. Build context
        let system = self.context_service.build_system_prompt(&self.config);
        let rag = self.context_service.inject_rag(&user_message).await;
        messages.push(Box::new(SystemMessage::new(system)));
        if let Some(ctx) = rag {
            messages.push(Box::new(SystemMessage::new(ctx)));
        }
        messages.extend(history);
        messages.push(Box::new(user_message));

        // 2. Loop
        let mut iteration = 0;
        let mut consecutive_text = 0;

        loop {
            iteration += 1;
            if iteration > self.config.max_iterations { break; }

            // 3. Call LLM
            let ai_msg = self.provider.chat_completion(&messages).await?;

            // 4. Stream text deltas
            for block in &ai_msg.content {
                if let ContentBlock::Text { text } = block {
                    self.event_tx.send(AgentEvent::TextDelta {
                        content: text.clone()
                    }).await?;
                }
            }

            // 5. Check for tool calls
            if ai_msg.tool_calls.is_empty() {
                consecutive_text += 1;
                if consecutive_text >= self.config.max_consecutive_text {
                    break; // Force stop
                }
                messages.push(Box::new(ai_msg));
                break;
            }

            consecutive_text = 0;
            messages.push(Box::new(ai_msg.clone()));

            // 6. Execute tools
            for tool_call in &ai_msg.tool_calls {
                self.event_tx.send(AgentEvent::ToolUse {
                    name: tool_call.name.clone(),
                    args: tool_call.args.clone(),
                }).await?;

                let runtime = self.build_tool_runtime(tool_call.id.clone());
                let result = self.tools.execute(tool_call.clone(), runtime).await?;

                self.event_tx.send(AgentEvent::ToolResult {
                    tool_call_id: tool_call.id.clone(),
                    output: result.content.clone(),
                    is_error: !result.success,
                }).await?;

                messages.push(Box::new(ToolMessage {
                    content: result.content,
                    tool_call_id: tool_call.id.clone(),
                    ..Default::default()
                }));
            }
        }

        // 7. Extract final response
        Ok(AgentEvent::Response {
            content: extract_final_text(&messages),
            tool_calls_made: iteration as u32,
            usage: self.calculate_usage(&messages),
        })
    }
}

6b. Response Filter Middleware

The single chokepoint where every event passes before reaching the frontend. Controls what the user sees.

Why This Matters

Currently, every AgentEvent flows from the agent loop → chat.rs → app.emit() → frontend with zero filtering. The model's raw output, internal tool details, and verbose reasoning all reach the user unchecked.

// ═══════════════════════════════════════════════════════════════
// RESPONSE FILTER — Single chokepoint for all events
// ═══════════════════════════════════════════════════════════════

/// Response filter trait — each filter is a composable layer
#[async_trait]
trait EventFilter: Send + Sync {
    /// Filter/transform an event. Returns None to suppress the event.
    async fn filter(&self, event: AgentEvent, ctx: &FilterContext) -> Option<AgentEvent>;
}

/// Context available to all filters
struct FilterContext {
    session_id: String,
    agent_id: String,
    project_path: PathBuf,
    config: FilterConfig,
}

struct FilterConfig {
    /// Max chars for tool results shown to user
    max_tool_result_display: usize,    // Default: 500

    /// Strip internal reasoning from display
    strip_reasoning: bool,             // Default: true

    /// Add cost summary to final response
    add_cost_summary: bool,            // Default: true

    /// Patterns to strip from text (API keys, tokens, etc.)
    sensitive_patterns: Vec<String>,

    /// Events to suppress entirely
    suppressed_events: Vec<String>,
}

/// The filter pipeline
struct ResponseFilterPipeline {
    filters: Vec<Arc<dyn EventFilter>>,
}

impl ResponseFilterPipeline {
    async fn apply(&self, event: AgentEvent, ctx: &FilterContext) -> Option<AgentEvent> {
        let mut current = Some(event);
        for filter in &self.filters {
            if let Some(evt) = current {
                current = filter.filter(evt, ctx).await;
            } else {
                return None; // Event suppressed
            }
        }
        current
    }
}

Concrete Filters

// ═══════════════════════════════════════════════════════════════
// FILTER 1: TextFilter — Clean up text deltas
// ═══════════════════════════════════════════════════════════════

struct TextFilter;

#[async_trait]
impl EventFilter for TextFilter {
    async fn filter(&self, event: AgentEvent, ctx: &FilterContext) -> Option<AgentEvent> {
        match event {
            AgentEvent::TextDelta { content } => {
                // Strip internal markers like [PLAN], [TODO], etc.
                let cleaned = content
                    .replace("[PLAN]", "")
                    .replace("[TODO]", "")
                    .replace("[DONE]", "")
                    .trim()
                    .to_string();

                if cleaned.is_empty() { None }
                else { Some(AgentEvent::TextDelta { content: cleaned }) }
            }
            _ => Some(event),
        }
    }
}

// ═══════════════════════════════════════════════════════════════
// FILTER 2: ToolFilter — Control what tool details show
// ═══════════════════════════════════════════════════════════════

struct ToolFilter {
    max_result_display: usize,
}

#[async_trait]
impl EventFilter for ToolFilter {
    async fn filter(&self, event: AgentEvent, ctx: &FilterContext) -> Option<AgentEvent> {
        match event {
            AgentEvent::ToolResult { tool_call_id, output, is_error } => {
                // Truncate long outputs for display
                let display_output = if output.len() > self.max_result_display {
                    format!("{}... ({} total chars)",
                        &output[..self.max_result_display],
                        output.len())
                } else {
                    output
                };
                Some(AgentEvent::ToolResult { tool_call_id, output: display_output, is_error })
            }
            AgentEvent::ToolUse { name, args } => {
                // Summarize args for display (don't show full JSON)
                Some(AgentEvent::ToolUse { name, args }) // Pass through, frontend handles display
            }
            _ => Some(event),
        }
    }
}

// ═══════════════════════════════════════════════════════════════
// FILTER 3: SensitiveFilter — Strip secrets/keys
// ═══════════════════════════════════════════════════════════════

struct SensitiveFilter {
    patterns: Vec<Regex>,
}

impl SensitiveFilter {
    fn new() -> Self {
        Self {
            patterns: vec![
                Regex::new(r"(?i)(api[_-]?key|token|secret|password)\s*[=:]\s*\S+").unwrap(),
                Regex::new(r"sk-[a-zA-Z0-9]{20,}").unwrap(),
                Regex::new(r"ghp_[a-zA-Z0-9]{36}").unwrap(),
            ],
        }
    }
}

#[async_trait]
impl EventFilter for SensitiveFilter {
    async fn filter(&self, event: AgentEvent, ctx: &FilterContext) -> Option<AgentEvent> {
        match event {
            AgentEvent::TextDelta { content } => {
                let mut filtered = content;
                for pattern in &self.patterns {
                    filtered = pattern.replace_all(&filtered, "[REDACTED]").to_string();
                }
                Some(AgentEvent::TextDelta { content: filtered })
            }
            AgentEvent::ToolResult { tool_call_id, output, is_error } => {
                let mut filtered = output;
                for pattern in &self.patterns {
                    filtered = pattern.replace_all(&filtered, "[REDACTED]").to_string();
                }
                Some(AgentEvent::ToolResult { tool_call_id, output: filtered, is_error })
            }
            _ => Some(event),
        }
    }
}
