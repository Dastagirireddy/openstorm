OpenStorm AI v2 — Frontend Architecture

Complete frontend redesign using Web Components, Lit.js 3.x, and composable custom elements. Follows industry standards from Cursor, VS Code Copilot, and modern AI chat UIs.

Design before implementation — all components, events, and flows defined for future coding.

1. Industry Standards & Research

Design Philosophy
We adopt proven patterns from the best AI coding tools (Cursor, VS Code Copilot, Claude Code) and modern WC libraries (Loquix, Chativa, Kitn AI). No re-invention — only integration of what works.

Patterns to Adopt

Float + custom layout modes. cds-aichat-container and cds-aichat-custom-element.

Key Lesson from Failures

VS Code Chat Performance Bug

Keeping all messages in DOM causes progressive lag. After hundreds of messages with code blocks, DOM grows to tens of thousands of nodes. Solution: Virtual scrolling — only render visible messages + buffer. Slack, Discord, and VS Code's own editor all use this pattern.

Three-Layer Component Model

Inspired by Kitn AI
Headless primitives → accessible UI → feature components. Each layer depends only on the one below it.
Layer 1: Headless Primitives
Logic-only, no visual output
  • useStreaming — streaming text controller
  • useVirtualScroll — virtual list logic
  • usePermission — approval state machine
  • useMessageStore — message CRUD
  • useAgentSpawner — sub-agent lifecycle
Layer 2: UI Primitives
Generic, reusable, accessible
  • <os-button>, <os-input>, <os-badge>
  • <os-popover>, <os-tooltip>
  • <os-collapsible>, <os-tabs>
  • <os-avatar>, <os-spinner>
  • <os-code-block>, <os-markdown>
Layer 3: AI Features
Domain-specific, composable
  • <openstorm-ai-panel>
  • <openstorm-ai-message-list>
  • <openstorm-ai-composer>
  • <openstorm-ai-permission-bar>
  • <openstorm-ai-tool-call>
Layout
Panel arrangement & resizing
  • <openstorm-ai-panel> — full / panel / floating
  • <openstorm-ai-split-view> — editor + chat

4. State Management

Central AI State Store

// ═══════════════════════════════════════════════════════════════
// AI STATE — Single source of truth
// ═══════════════════════════════════════════════════════════════

interface AIState {
  // ── Session ──
  sessionId:          string;
  conversationId:     string;

  // ── Messages ──
  messages:           AIMessage[];
  streamingMessage:   AIMessage | null;

  // ── Agent ──
  isStreaming:        boolean;
  isThinking:         boolean;
  currentModel:       string;
  agentId:            string | null;

  // ── Tools ──
  pendingApprovals:   ToolApproval[];
  activeToolCalls:    ToolCall[];

  // ── Sub-agents ──
  subAgents:          SubAgent[];

  // ── Questions ──
  pendingQuestions:   Question[];

  // ── Context ──
  projectPath:        string;
  attachedFiles:      string[];
  ragContext:         string | null;

  // ── Metrics ──
  totalTokens:        number;
  totalCost:          number;
  lastLatencyMs:      number;
}

interface AIMessage {
  id:         string;
  role:       'user' | 'assistant' | 'system' | 'tool';
  content:    string;
  timestamp:  number;
  toolCalls?: ToolCall[];
  metadata?:  MessageMetadata;
  streaming?: boolean;
}

Component Subscription Pattern

// Components receive state via properties
@customElement('openstorm-ai-panel')
export class AIPanel extends TailwindElement {
  @state() private state: AIState = DEFAULT_STATE;

  connectedCallback() {
    super.connectedCallback();
    // Subscribe to central store
    aiStore.subscribe('messages', (msgs) => {
      this.state = { ...this.state, messages: msgs };
    });
    aiStore.subscribe('isStreaming', (val) => {
      this.state = { ...this.state, isStreaming: val };
    });
  }
}

All cross-component communication via typed CustomEvents. No direct method calls between components. Events use bubbles: true, composed: true to cross Shadow DOM boundaries.

<openstorm-ai-message-list> Performance

Virtual Scrolling Required
This component MUST use virtual scrolling. Only visible messages + buffer are in the DOM at any time. See Section 7 for implementation details. 
// ═══════════════════════════════════════════════════════════════
// AI MESSAGE LIST — Virtual-scrolled message container
// ═══════════════════════════════════════════════════════════════

@customElement('openstorm-ai-message-list')
export class AIMessageList extends TailwindElement {
  @property({ type: Array })  messages: AIMessage[] = [];
  @property({ type: Object }) streamingMessage: AIMessage | null = null;
  @query('.scroll-container') private scrollContainer!: HTMLElement;

  private virtualScroll: VirtualScrollController;

  connectedCallback() {
    super.connectedCallback();
    this.virtualScroll = new VirtualScrollController({
      itemHeight: 80,       // estimated
      buffer: 5,            // extra items above/below
      container: () => this.scrollContainer,
      renderItem: (msg) => this.renderMessage(msg),
    });
  }

  updated(changed) {
    if (changed.has('messages')) {
      this.virtualScroll.setItems(this.messages);
      this.autoScroll();
    }
  }

  private renderMessage(msg: AIMessage) {
    return html\`
      <openstorm-ai-message
        .message=\${msg}
        @ai:approve-tool=\${this.onApprove}
      ></openstorm-ai-message>
    `;
  }

  render() {
    return html\`
      <div class="scroll-container flex-1 overflow-y-auto px-4 py-3">
        \${this.virtualScroll.render()}
        \${this.streamingMessage
          ? html\`<openstorm-ai-message
              .message=\${this.streamingMessage}
              .streaming=\${true}
            ></openstorm-ai-message>\`
          : nothing}
        <div class="scroll-anchor"></div>
      </div>
    `;
  }
}

@customElement('openstorm-ai-message')
export class AIMessage extends TailwindElement {
  @property({ type: Object }) message: AIMessage | null = null;
  @property({ type: Boolean }) streaming: boolean = false;

  render() {
    if (!this.message) return nothing;
    const isUser = this.message.role === 'user';

    return html\`
      <div class="flex gap-3 py-3 \${isUser ? 'flex-row-reverse' : ''}">
        <openstorm-ai-avatar .role=\${this.message.role} size="sm">
        </openstorm-ai-avatar>

        <div class="flex flex-col \${isUser ? 'items-end' : 'items-start'} max-w-[85%]">
          <div class="text-xs text-[var(--os-text-subtle)] mb-1">
            \${isUser ? 'You' : 'AI'} · \${this.formatTime(this.message.timestamp)}
          </div>

          <div class="rounded-lg px-3 py-2 \${isUser
            ? 'bg-[var(--os-ai-user-bg)] text-[var(--os-ai-user-text)]'
            : 'bg-[var(--os-ai-agent-bg)] text-[var(--os-ai-agent-text)]'}">

            <openstorm-ai-content
              .content=\${this.message.content}
              .streaming=\${this.streaming}
            ></openstorm-ai-content>

            \${this.message.toolCalls?.map(tc => html\`
              <openstorm-ai-tool-call .toolCall=\${tc} class="mt-2">
              </openstorm-ai-tool-call>
            \`)}
          </div>

          \${!isUser && !this.streaming ? html\`
            <openstorm-ai-actions
              .messageId=\${this.message.id}
              .content=\${this.message.content}
              class="mt-1"
            ></openstorm-ai-actions>
          \` : nothing}
        </div>
      </div>
    `;
  }
}

// ═══════════════════════════════════════════════════════════════
// VIRTUAL SCROLL CONTROLLER — Headless logic for virtual lists
// ═══════════════════════════════════════════════════════════════

class VirtualScrollController {
  private items: any[] = [];
  private visibleRange = { start: 0, end: 50 };
  private itemHeight: number;
  private buffer: number;

  constructor(config: { itemHeight: number; buffer: number }) {
    this.itemHeight = config.itemHeight;
    this.buffer = config.buffer;
  }

  // Only return visible items + buffer
  getVisibleItems(): any[] {
    return this.items.slice(
      Math.max(0, this.visibleRange.start - this.buffer),
      Math.min(this.items.length, this.visibleRange.end + this.buffer)
    );
  }

  // Update range on scroll
  onScroll(scrollTop: number, containerHeight: number) {
    const start = Math.floor(scrollTop / this.itemHeight);
    const end = start + Math.ceil(containerHeight / this.itemHeight);
    this.visibleRange = { start, end };
  }

  // Spacer heights for correct scroll position
  getTopSpacer(): number {
    return Math.max(0, this.visibleRange.start - this.buffer) * this.itemHeight;
  }

  getBottomSpacer(): number {
    return Math.max(0, this.items.length - this.visibleRange.end - this.buffer)
      * this.itemHeight;
  }
}

import { invoke } from '@tauri-apps/api/core';
import { listen } from '@tauri-apps/api/event';

// ── Commands (Frontend → Backend) ──
export const aiCommands = {
  sendMessage: (msg, history, project, model) =>
    invoke('ai_v2_chat', {
      request: { message: msg, history, project_path: project, model }
    }),

  approveTool: (toolCallId, approved) =>
    invoke('ai_v2_approve_tool', {
      request: { tool_call_id: toolCallId, approved }
    }),

  abort: () =>
    invoke('ai_v2_abort'),

  spawnAgent: (task, role) =>
    invoke('ai_v2_spawn_agent', {
      request: { task, role, parent_id: null }
    }),

  answerQuestion: (answers) =>
    invoke('ai_v2_question_response', {
      request: { answers }
    }),
};

// ── Events (Backend → Frontend) ──
export function listenToAgentEvents(dispatch) {
  const unlisteners = [
    listen('ai-v2:text-delta', (e) =>
      dispatch({ type: 'text-delta', ...e.payload })),
    listen('ai-v2:tool-use', (e) =>
      dispatch({ type: 'tool-use', toolCall: e.payload })),
    listen('ai-v2:tool-result', (e) =>
      dispatch({ type: 'tool-result', ...e.payload })),
    listen('ai-v2:tool-approval-required', (e) =>
      dispatch({ type: 'permission-request', approval: e.payload })),
    listen('ai-v2:question-request', (e) =>
      dispatch({ type: 'question-asked', questions: e.payload })),
    listen('ai-v2:error', (e) =>
      dispatch({ type: 'error', message: e.payload.message })),
  ];
  return () => unlisteners.forEach(unlisten => unlisten());
}
